. program define build_vars
  1.     import delimited "panel_with_broad_tax.csv", encoding(UTF-8) clear
  2.     gen pub_emp_share2 = emp_urban_unit_10k / (emp_urban_unit_10k + private_emp_10k)
  3.     gen labor          = emp_urban_unit_10k + private_emp_10k
  4.     scalar delta = 0.096
  5.     scalar alpha = 0.5
  6.     scalar g_inv = 0.161
  7.     xtset province_id year
  8.     gen capital = .
  9.     levelsof province_id, local(provinces)
 10.     foreach p of local provinces {
 11.         quietly sum fai_bn if province_id == `p' & year == 2008
 12.         local I2008 = r(mean)
 13.         quietly replace capital = `I2008' / (g_inv + delta) ///
>             if province_id == `p' & year == 2007
 14.     }
 15.     sort province_id year
 16.     foreach y of numlist 2008/2017 {
 17.         replace capital = (1 - delta) * L.capital + fai_bn ///
>             if year == `y' & fai_bn != .
 18.     }
 19.     gen ln_gdp_val = ln(gdp_bn)
 20.     gen ln_capital = ln(capital)
 21.     gen ln_labor   = ln(labor)
 22.     gen ln_pub_emp = ln(pub_emp_share2)
 23.     gen ln_aging   = ln(aging_rate)
 24.     gen ln_urban   = ln(urban_rate)
 25.     gen ln_trade   = ln(trade_openness) if trade_openness > 0
 26.     gen ln_tfp     = ln_gdp_val - 0.5*ln_capital - 0.5*ln_labor
 27.     gen ln_tfp_alt = ln_gdp_val - 0.4*ln_capital - 0.6*ln_labor
 28.     gen tech_density    = (pub_tech_staff / gdp_bn) * 1000
 29.     gen ln_tech_density = ln(tech_density)
 30.     gen L3_pub_emp = L3.pub_emp_share2
 31.     gen L5_pub_emp = L5.pub_emp_share2
 32.     gen L3_aging   = L3.aging_rate
 33.     gen L5_aging   = L5.aging_rate
 34.     gen sample_main = (year >= 2008 & year <= 2017) & capital != . & ln_tfp != .
 35.     gen sample_lag  = (year >= 2011 & year <= 2017) ///
>         & L3_pub_emp != . & L3_aging != . & ln_tfp != .
 36.     gen sample_l5   = (year >= 2013 & year <= 2017) ///
>         & L5_pub_emp != . & L5_aging != . & ln_tfp != .
